\begin{lstlisting}
int function1(int* a, int n) {
    int result = 0;
    for (int i = 0; i < n; ++i) {
        int x = 5;
        for (int j = 0; j < n; ++j) {
            if (j % 2 == 0)
                continue;
            result += 2 * 3 * a[j] * x - i - 2;
        } 
    }

    return result;
}
\end{lstlisting}

\paragraph{Что делает:} Аргументами принимает указатель на int $a$ и целое $n$, возвращает $int$  $result$. Эта программа проходит внешним циклом от 0 до $n$ и внутренним от 0 до $n$ по нечетным значениям счетчика $j$. К переменной $result$ прибавляется $2 * 3 * a[j] * x - i - 2$.

\subsection{Do-While представление}
\begin{lstlisting}
int function1(int* a, int n) {
    int result = 0;

    int i = 0;
    if (i >= n) {
        return result;
    }

    do {
        int x = 5;

        int j = 0;
        if (j < n) {
            do {
                if (j % 2 == 0) {
                    j++;
                    continue;
                }
                
                result += 2 * 3 * a[j] * x - i - 2;
                j++;
            } while (j < n);
        }
        
        i++;
    } while (i < n);

    return result;
}
\end{lstlisting}

int function1(int* a, int n) {
  func_start:
    result = 0;
    i = 0;

    loop1_neg_cnd = (i >= n);
    br loop1_neg_cnd, exit, loop1_body;

  loop1_body:
    x = 5;
    j = 0;

    loop2_cnd = (j < n);
    br loop2_cnd, loop2_head, loop1_end;

  loop2_head:
    e = (j % 2);
    f = (e == 0);
    br f, loop2_end, loop2_body;

  loop2_body:
    tmp = 2 * 3;
        
    addr = a + j;
    val = *addr;

    tmp = tmp * val;
    tmp = tmp * x;
    tmp = tmp - i;
    tmp = tmp - 2;
    result = result + tmp;
    br loop2_end;

  loop2_end:
    j = j + 1;

    loop2_cnd = (j < n);
    br loop2_cnd, loop2_head, loop1_end;

  loop1_end:
    i = i + 1;
    loop1_cnd = (i < n);
    br loop1_cnd, loop1_body, exit;
    
  exit:
    return result;
}
\end{lstlisting}

\subsection{Constant propagation + Алгебраическое переупорядочивание + Constant folding}

\subsection{Loop Strength Resuction}
Заметим, что в внутреннем цикле индуктивная переменная $\texttt{j} = <0, +, 1>$, при этом внутри стоит проверка на $\texttt{j} \% 2 == 0$. В таком случае можем заменить индуктивность на $\texttt{j} = <1, +, 2>$, а лишнюю проверку выкинуть.
